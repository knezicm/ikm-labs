# 1. Load vcan module
sudo modprobe vcan

# 2. Confirm that vcan module is loaded
lsmod | grep vcan

# You should get something like this
vcan                   12288  0
can_dev                40960  1 vcan

# 3. Initialize the vcan interface
sudo ip link add dev vcan0 type vcan
sudo ip link set vcan0 mtu 16
sudo ip link set up vcan0

# 4. Confirm that vcan0 interface is up and running
ifconfig vcan0
